def strlen(string): length = 0 for char in string: length += 1 return length def strcpy(source, destination): for i in range(len(source)): destination[i] = source[i] def strcat(string1, string2): return string1 + string2 def strstr(string, substring): for i in range(len(string) - len(substring) + 1): if string[i:i+len(substring)] == substring: return i return -1 def strrchr(string, char): index = -1 for i in range(len(string)): if string[i] == char: index = i return index string = "Hello, World!" substring = "World" char = "l" print(strlen(string)) destination = [''] * len(string) strcpy(string, destination) print(''.join(destination)) print(strcat(string, substring)) print(strstr(string, substring)) print(strrchr(string, char))
