def search_element(array, element): for i in range(len(array)): if array[i] == element: return i return -1 array = list(map(int, input("Enter the elements of the array separated by spaces: ").split())) element = int(input("Enter the element to search for: ")) position = search_element(array, element) if position != -1: print(f"Element {element} found at position {position}") else: print(f"Element {element} not found in the array")
