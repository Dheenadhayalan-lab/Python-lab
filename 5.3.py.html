def find_largest_and_smallest(numbers): if not numbers: return None, None largest = numbers[0] smallest = numbers[0] for num in numbers[1:]: if num > largest: largest = num if num < smallest: smallest = num return largest, smallest numbers_input = input("Enter numbers separated by spaces: ") numbers = list(map(float, numbers_input.split())) largest, smallest = find_largest_and_smallest(numbers) if largest is not None and smallest is not None: print(f"The largest number is: {largest}") print(f"The smallest number is: {smallest}") else: print("No numbers were entered.")
